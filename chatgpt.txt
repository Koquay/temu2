When a user is signed in on the server, a json web token is constructed as a token and communicated to the client.
When the user logs off, how do I disable the token on the server?

const getToken = (userId) => {
    let token = jwt.sign(
      {
        userId: userId
      },
      process.env.JWT_SECRET,
      {
        expiresIn: "2m",
      }
    );
  
    return token;
  }
